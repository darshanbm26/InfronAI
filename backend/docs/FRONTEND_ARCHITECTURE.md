## API Integration

### API Client (`src/lib/api.ts`)

```typescript
import axios from 'axios';

const API_BASE = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:8000';

export const api = {
  // Start analysis (Phase 1-6)
  startAnalysis: async (description: string) => {
    const response = await axios.post(`${API_BASE}/api/v1/analyze`, {
      description,
    });
    return response.data; // { session_id: "..." }
  },

  // Get analysis status (SSE endpoint)
  getAnalysisStream: (sessionId: string) => {
    return new EventSource(`${API_BASE}/api/v1/analyze/${sessionId}/stream`);
  },

  // Get recommendation presentations
  getPresentation: async (sessionId: string, type: 'executive' | 'detailed' | 'technical') => {
    const response = await axios.get(
      `${API_BASE}/api/v1/recommendation/${sessionId}/presentation?type=${type}`
    );
    return response.data;
  },

  // Submit decision (Phase 7)
  submitDecision: async (params: {
    sessionId: string;
    decisionType: 'accepted' | 'customized' | 'rejected';
    customizationDetails?: object;
    userFeedback?: string;
    decisionTimeSeconds: number;
  }) => {
    const response = await axios.post(`${API_BASE}/api/v1/decision`, {
      session_id: params.sessionId,
      decision_type: params.decisionType,
      customization_details: params.customizationDetails,
      user_feedback: params.userFeedback,
      decision_time_seconds: params.decisionTimeSeconds,
    });
    return response.data; // { decision_id: "...", artifacts: {...}, next_actions: [...] }
  },

  // Get artifacts
  getArtifacts: async (decisionId: string) => {
    const response = await axios.get(`${API_BASE}/api/v1/artifacts/${decisionId}`);
    return response.data;
  },

  // Download all artifacts as ZIP
  downloadArtifactsZip: async (decisionId: string) => {
    const response = await axios.get(
      `${API_BASE}/api/v1/artifacts/${decisionId}/download`,
      { responseType: 'blob' }
    );
    return response.data;
  },
};
```

## Real-time Updates (SSE)

The analysis page uses Server-Sent Events to show live progress:

```typescript
// src/hooks/useSSE.ts
import { useEffect, useState } from 'react';

export function useAnalysisStream(sessionId: string) {
  const [phases, setPhases] = useState<PhaseStatus[]>([]);
  const [isComplete, setIsComplete] = useState(false);

  useEffect(() => {
    const eventSource = new EventSource(
      `${API_BASE}/api/v1/analyze/${sessionId}/stream`
    );

    eventSource.onmessage = (event) => {
      const data = JSON.parse(event.data);
      
      if (data.type === 'phase_update') {
        setPhases(prev => [...prev, data.phase]);
      } else if (data.type === 'complete') {
        setIsComplete(true);
        eventSource.close();
      }
    };

    return () => eventSource.close();
  }, [sessionId]);

  return { phases, isComplete };
}
```

## Performance Optimizations for End User

1. **Streaming Updates**: Don't wait for all phases - show progress in real-time
2. **Skeleton Loading**: Show UI structure immediately, fill in data as it arrives
3. **Optimistic UI**: When user clicks ACCEPT, show success immediately while API processes
4. **Caching**: Cache results by session_id for back-navigation
5. **Code Splitting**: Lazy load heavy components (charts, code viewers)
